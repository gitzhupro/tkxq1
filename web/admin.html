    <h1>可视化编辑 data.json</h1>

    <p>下方文本框内直接修改 JSON，保存后立即生效（GitHub Pages 需重新推送）。</p>

    <textarea id="txt"></textarea>

    <br>

    <button onclick="save()">保存到本地文件</button>

    <span id="tip"></span>


    <script>

        const el = document.getElementById('txt');

        const tip = document.getElementById('tip');


        // 初始载入

        fetch('data.json').then(r=>r.text()).then(t=>el.value=t);


        function save(){

            try{

                JSON.parse(el.value); // 校验

                const blob = new Blob([el.value],{type:'application/json'});

                const a = document.createElement('a');

                a.download = 'data.json';

                a.href = URL.createObjectURL(blob);

                a.click();

                tip.textContent = '下载成功！用新生成的 data.json 覆盖仓库同名文件并 push 即可。';

                tip.className = 'ok';

            }catch(e){

                tip.textContent = 'JSON 格式错误：'+e.message;

                tip.className = 'err';

            }

        }

    </script>
