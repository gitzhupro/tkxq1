<script>

    /* ========== 单页导航 ========== */

    document.querySelectorAll('nav li').forEach(el=>{

        el.onclick = ()=>{

            const tar = document.getElementById(el.dataset.to);

            tar && tar.scrollIntoView({behavior:'smooth'})

        }

    });


    /* ========== 读取数据并渲染 ========== */

    fetch('data.json').then(r=>r.json()).then(renderAll);


    function renderAll(D){

        // 首屏

        document.getElementById('heroTitle').textContent = D.hero.title;

        document.getElementById('heroDesc').textContent = D.hero.desc;


        // 服务

        const sg = document.getElementById('serviceGrid');

        D.services.forEach(o=>{

            sg.insertAdjacentHTML('beforeend',`

            <div class="card">

                <div class="txt">

                    <h3>${o.title}</h3>

                    <p>${o.summary}</p>

                </div>

            </div>`);

        });


        // 作品

        const wb = document.getElementById('worksBox');

        D.works.forEach(o=>{

            wb.insertAdjacentHTML('beforeend',`<img src="${o.img}" alt="${o.title}">`);

        });


        // 文章

        const bg = document.getElementById('blogGrid');

        D.posts.forEach(o=>{

            bg.insertAdjacentHTML('beforeend',`

            <div class="card">

                <img src="${o.img}" alt="${o.title}">

                <div class="txt">

                    <h3>${o.title}</h3>

                    <p>${o.summary}</p>

                </div>

            </div>`);

        });

    }

</script>
